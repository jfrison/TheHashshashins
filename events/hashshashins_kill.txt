						limit = { check_variable = { which = hashshashin_assassination_chance value = 50 } }

					}

						set_country_flag = hashshashin_assassination_succeeded
					}
				}
			}
		
			if = {
				limit = { has_country_flag = hashshashin_assassination_succeeded }
